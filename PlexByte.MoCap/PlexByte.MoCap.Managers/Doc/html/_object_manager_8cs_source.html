597  private T CreateObjectFromForm<T>(DockContent pForm)
598  {
599  if (pForm.GetType() == typeof(uc_Project))
600  {
601  IProject tmp = _formManager.CreateObjectFromForm<IProject>(pForm);
602  ProjectList.Add(tmp);
603  _dataManager.UpsertProject((Project)tmp);
604  UpsertProjectMapping(tmp, tmp.Creator.Id, tmp.Creator);
605  return (T)tmp;
606  }
607  else if (pForm.GetType() == typeof(uc_User))
608  {
609  IUser tmp = _formManager.CreateUserObjectFromForm((uc_User)pForm);
610  if (!UserList.Contains(tmp))
611  UserList.Add(tmp);
612  _dataManager.UpsertUser((User)tmp);
613  return (T)tmp;
614  }
615  else if (pForm.GetType() == typeof(uc_Task))
616  {
617  ITask tmp = _formManager.CreateObjectFromForm<ITask>(pForm);
618  TaskList.Add(tmp);
619  _dataManager.UpsertTask((Task)tmp);
620  return (T)tmp;
621  }
622  else if (pForm.GetType() == typeof(uc_Survey))
623  {
624  ISurvey tmp = _formManager.CreateObjectFromForm<ISurvey>(pForm);
625  SurveyList.Add(tmp);
626  _dataManager.UpsertSurvey((Survey)tmp);
627  foreach (ISurveyOption option in tmp.OptionList)
628  _dataManager.UpsertSurveyOption((SurveyOption)option, tmp.Id);
629  tmp.VoteList = _dataManager.GetVoteBySurveyId(tmp.Id);
630  return (T)tmp;
631  }
632  else if (pForm.GetType() == typeof(uc_Expense))
633  {
634  IExpense tmp = _formManager.CreateObjectFromForm<IExpense>(pForm);
635  ExpenseList.Add(tmp);
636  _dataManager.UpsertExpense((Expense)tmp);
637  return (T)tmp;
638  }
639  else if (pForm.GetType() == typeof(uc_Timeslice))
640  {
641  ITimeslice tmp = _formManager.CreateObjectFromForm<ITimeslice>(pForm);
642  TimesliceList.Add(tmp);
643  _dataManager.UpsertTimeslice((Timeslice)tmp);
644  return (T)tmp;
645  }
646  else
647  {
648  throw new InvalidCastException($"The type {pForm.GetType().ToString()} cannot be created as was never defined");
649  }
