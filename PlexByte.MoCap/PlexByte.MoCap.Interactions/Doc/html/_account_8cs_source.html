360  {

362  {
363  // Behind schedule?
364  if (EndDateTime <= DateTime.Now)
365  ChangeState(InteractionState.Expired);
366  // Turns active?
367  if (StartDateTime <= DateTime.Now && State == InteractionState.Queued)
368  ChangeState(InteractionState.Active);
369  // Finished if project is closed?
370  if (IsActive == false)
371  ChangeState(InteractionState.Finished);
372  }
373 
374  // Still active?

376  _stateTimer.Start();
