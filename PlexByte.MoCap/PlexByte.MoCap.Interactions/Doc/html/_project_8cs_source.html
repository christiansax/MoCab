421  private void OnTimerElapsed(object sender, System.Timers.ElapsedEventArgs e)
422  {
423  if (State == InteractionState.Active || State == InteractionState.Queued)
424  {
425  // Behind schedule?
426  if (EndDateTime <= DateTime.Now)
427  ChangeState(InteractionState.Expired);
428  // Turns active?
429  if (StartDateTime <= DateTime.Now && State == InteractionState.Queued)
430  ChangeState(InteractionState.Active);
431  // Finished if project is closed?
432  if (IsActive == false)
433  ChangeState(InteractionState.Finished);
434  }
435 
436  // Still active?
437  if (State == InteractionState.Active || State == InteractionState.Queued)
438  _stateTimer.Start();
439  }
